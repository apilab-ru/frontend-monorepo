# UserAvatar
Аватар пользователя, для использования необходимо подключить LayoutModule

```
<app-user-avatar [user]="user" (openProfile)="onOpenProfile()"></app-user-avatar>
```

export const Desktop = args => ({
  template: `<app-user-avatar [user]="user"></app-user-avatar>`,
  props: {
    user: {
      photo: args.photo,
    }
  }
})
